Bağlantıyı test edip ekleyebiliriz.

Veritabanı kontrolü için SQL Lab örnek sorgu kodu;


SELECT 
    c.category_name AS "Kategori Adı",
    SUM(od.quantity) AS "Toplam Satılan Ürün Adeti",
    ROUND(SUM(od.unit_price * od.quantity * (1 - od.discount))::numeric, 2) AS "Toplam Gelir ($)"
FROM 
    categories c
JOIN 
    products p ON c.category_id = p.category_id
JOIN 
    order_details od ON p.product_id = od.product_id
GROUP BY 
    c.category_name
ORDER BY 
    "Toplam Gelir ($)" DESC;
